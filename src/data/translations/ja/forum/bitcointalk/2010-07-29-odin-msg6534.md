# これで現在インストールされているバージョンが表示される

# おそらくglibcバージョン2.10以前がインストールされているのだろう。

apt-cache showpkg libc6 | less
# （下の方で）2.11バージョンが利用可能か確認する

#単純にアップグレードできるかもしれない：
apt-get upgrade libc6
